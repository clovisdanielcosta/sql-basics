Dim asFinais$

Sub ConverterSQL()

    Dim i%
    aspasFinais = """"
    i = 4
    
    While WorksheetFunction.CountA(pConversor.Range(Cells(i, 5), Cells(i + 5, 8))) <> 0
    
                For j = 6 To 11
                    pConversor.Cells(i, 5).Value = UCase(pConversor.Cells(i, 5).Value) & " " & UCase(pConversor.Cells(i, j).Value)
                Next
                
                pConversor.Cells(i, 5).Value = aspasFinais & " " & LTrim(pConversor.Cells(i, 5).Value)
                pConversor.Cells(i, 5).Value = RTrim(pConversor.Cells(i, 5).Value) & " " & aspasFinais & " & _"
                pConversor.Range(Cells(i, 6), Cells(i, 11)).ClearContents
                
                i = i + 1
    Wend
    
    pConversor.Cells(i - 1, 5).Value = Replace(pConversor.Cells(i - 1, 5).Value, "& _", "")
            
End Sub

Sub CarregarExemplo()

    Range("S4:Y17").Select
    Selection.Copy
    Range("E4").Select
    ActiveSheet.Paste
    Range("E4").Select
    Application.CutCopyMode = False
    
End Sub

Sub SubstituirVariaveis()

    Dim variavel$, substituto$, direita$, esquerda$
    Dim i%, j%
    
    aspasFinais = """"
    esquerda = " '  " & aspasFinais & " & "
    direita = " & " & aspasFinais & " ' "
    i = 4
    
    While pConversor.Cells(i, 2).Value <> ""
        
            variavel = UCase(pConversor.Cells(i, 2).Value)
            substituto = UCase(pConversor.Cells(i, 3).Value)
            
                        j = 4
                        While pConversor.Cells(j, 5).Value <> ""
                                
                                If Left(variavel, 1) = "'" Then
                                    pConversor.Cells(j, 5).Value = Replace(pConversor.Cells(j, 5).Value, variavel, esquerda & substituto & direita)
                                Else
                                    pConversor.Cells(j, 5).Value = Replace(pConversor.Cells(j, 5).Value, variavel, substituto)
                                End If
                                
                                j = j + 1
                        
                        Wend
            
            i = i + 1
            
    Wend
        
End Sub
